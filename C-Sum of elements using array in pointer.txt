#include <stdio.h>
int main() {
    int n, i, sum = 0;
    int arr[100];
    int *ptr;
    printf("Enter number of elements: ");
    scanf("%d", &n);
    ptr = arr;
    printf("Enter %d integers:\n", n);
    for (i = 0; i < n; i++) {
        scanf("%d", ptr); 
        ptr++;              
    }
    ptr = arr;
    for (i = 0; i < n; i++) {
        sum += *ptr;       
        ptr++;  
    }
    printf("Sum of elements = %d\n", sum);
    return 0;
}
