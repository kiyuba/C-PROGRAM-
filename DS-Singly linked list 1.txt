#include<stdio.h>
#include<stdlib.h>
struct contact{
    long number;
    struct contact*nextwrd;
};
int main()
{
   
    struct contact*phno1=(struct contact*)malloc(sizeof(struct contact));
     struct contact*phno2=(struct contact*)malloc(sizeof(struct contact));
     struct contact*phno3=(struct contact*)malloc(sizeof(struct contact));
     phno1->number;
     phno2->number;
     phno3->number;
     printf("Enter The 1st Number: ");
     scanf("%d",&phno1->number);
     printf("Enter The 2nd Number :");
     scanf("%d",&phno2->number);
     printf("Enter The 3rd Number:");
     scanf("%d",&phno3->number);
     //inter connect
     phno1->nextwrd=phno2;
     phno2->nextwrd=phno3;
     phno3->nextwrd=NULL;
     struct contact*head=phno1;
    printf("\n");
     while(head!=0){
         printf("%ld-->",head->number);
         head=head->nextwrd;
     }
     printf("NULL");
     return 0;
     
     }
